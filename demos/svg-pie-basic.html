A basic SVG Pie chart

This goes in the documents header: 
<script src="RGraph.svg.common.core.js"></script>
<script src="RGraph.svg.pie.js"></script>
Put this where you want the chart to show up: 
<div style="width: 650px; height: 350px" id="chart-container"></div>
This is the code that generates the chart: 
<script>
    pie = new RGraph.SVG.Pie({
        id: 'chart-container',
        data: [85,48,13,56,79,95],
        options: {
            labels: ['Alf','Berty','Craig','Dean','Elgar','Fliss'],
            shadow: true,
            title: 'A basic Pie chart',
            exploded: [,25]
        }
    }).draw();
</script>
